LINE:
    | FUNCALL = EXPR                        # function definition
    | IDENTIFIER = EXPR                     # variable definition
    | EXPR                                  # some other expression

EXPR:
    | OPERATION                             # a + b     1 + 2
    | FUNCALL
    | PRIMITIVE
    | MATRIX
    | ARRAY
    | \(EXPR\)

OPERAND:
    | FUNCALL
    | PRIMITIVE
    | ARRAY
    | MATRIX

OPERATION:
    | OPERAND OPERATOR EXPR

FUNCALL:
    | IDENTIFIER\(\)
    | IDENTIFIER\(EXPR\)
    | IDENTIFIER\(EXPR(, EXPR)+\)

ARRAY:
    | \[EXPR\]
    | \[EXPR(?:, EXPR)*\]

MATRIX:
    | \[ARRAY\]
    | \[ARRAY(?:;ARRAY)*\]

PRIMITIVE:
    | IDENTIFIER
    | COMPLEX
    | FLOAT
    | INT
    | \(PRIMITIVE\)

# Primitives

IDENTIFIER:
    | LETTER (?:DIGIT|SYMBOL|LETTER)+

INT:
    | -?[0-9]+

FLOAT:
    | INT \.[0-9]*

COMPLEX:
    | (FLOAT|INT) i
    | i
